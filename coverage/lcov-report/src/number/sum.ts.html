/**
 * 计算一组数字的总和
 * 
 * @param numbers - 要计算总和的数字数组
 * @param options - 配置选项
 * @param options.ignoreNaN - 是否忽略非数字值，默认为true
 * @param options.precision - 结果精度（小数位数），默认不限制
 * @returns 总和
 * @example
 * ```ts
 * // 基本用法
 * sum([1, 2, 3, 4, 5]); // 返回: 15
 * 
 * // 带小数
 * sum([1.1, 2.2, 3.3]); // 返回: 6.6
 * 
 * // 设置精度
 * sum([1.11, 2.22, 3.33], { precision: 2 }); // 返回: 6.66
 * 
 * // 处理非数字
 * sum([1, 2, 'a', 3], { ignoreNaN: true }); // 返回: 6
 * sum([1, 2, 'a', 3], { ignoreNaN: false }); // 返回: NaN
 * 
 * // 空数组
 * sum([]); // 返回: 0
 * ```
 */
export function sum(
  numbers: Array<number | string | null | undefined>,
  options: { ignoreNaN?: boolean; precision?: number } = {}
): number {
  // 设置默认选项
  const { ignoreNaN = true, precision } = options;
  
  // 过滤并转换输入值
  const validNumbers = numbers
    .map(num => {
      // 尝试将输入转换为数字
      const parsedNumber = parseFloat(String(num));
      return isNaN(parsedNumber) ? NaN : parsedNumber;
    })
    .filter(num => ignoreNaN ? !isNaN(num) : true);
  
  // 如果没有有效数字或ignoreNaN为false且包含NaN
  if (validNumbers.length === 0 || validNumbers.some(isNaN)) {
    return validNumbers.length === 0 ? 0 : NaN;
  }
  
  // 计算总和
  const total = validNumbers.reduce((acc, val) => acc + val, 0);
  
  // 根据精度处理结果
  if (precision !== undefined && precision >= 0) {
    const factor = Math.pow(10, precision);
    return Math.round(total * factor) / factor;
  }
  
  return total;
}
