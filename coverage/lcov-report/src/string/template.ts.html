/**

 * @param template - 模板字符串

function compileTemplate(template: string): (data: Record<string, any>) => string {
  // 替换模板标记为JavaScript代码
  const code = template
    .replace(/\\/g, '\\\\')  // 转义反斜杠
    .replace(/"/g, '\\"')    // 转义双引号
    .replace(/\n/g, '\\n')   // 转义换行符
    .replace(/\r/g, '\\r')   // 转义回车符
    .replace(/\{\{([\s\S]+?)\}\}/g, '" + ($1) + "');  // 替换表达式
  
  // 构建函数体
  const fnBody = `
    try {
      return "${code}";

      return "模板解析错误: " + e.message;
    }
  `;
  
  // 创建一个函数，将数据对象的属性作为局部变量
  return function(data: Record<string, any>): string {
    const keys = Object.keys(data);
    const values = keys.map(key => data[key]);
    
    // 使用 Function 构造函数创建新函数，绑定数据
    const fn = new Function(...keys, fnBody);
