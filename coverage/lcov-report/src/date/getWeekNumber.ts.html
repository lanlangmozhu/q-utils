/**
 * 获取日期在一年中的第几天
 * 
 * @param date - 要计算的日期
 * @returns 天数（1-366）
 */
function getDayOfYear(date: Date): number {
  const start = new Date(date.getFullYear(), 0, 0);
  const diff = date.getTime() - start.getTime();
  const oneDay = 1000 * 60 * 60 * 24;
  return Math.floor(diff / oneDay);
}
